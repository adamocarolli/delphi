   class RV 
   {
     public:
       string name;
       double value;
       vector< double > dataset;
   
   
       RV( string name ) : name( name ) { }
   
   
       double sample()
       {
         vector< double > s;
   
         std::sample( this->dataset.begin(), this->dataset.end(), std::back_inserter( s ), 1, std::mt19937{ std::random_device{}() });
   
         return s[0];
       }
   
   };
